import numpy as np
import matplotlib.pyplot as plt
from tensorflow.keras.preprocessing.image import ImageDataGenerator
from tensorflow.keras.applications import VGG16
from tensorflow.keras.layers import Dense, Flatten
from tensorflow.keras.models import Model
from tensorflow.keras.optimizers import Adam
from tensorflow.keras.callbacks import EarlyStopping, ModelCheckpoint

# Define the size to which images will be resized
img_size = 224
# Set the number of images to process in each batch
batch_size = 16

# Path to the directory containing the images
dataset_path = "/home/cs/Desktop/MCA_RiceClassification_VGG16/Rice_Image_Dataset"  # Update this with your dataset path

# Create a data generator to preprocess images and split them into training and validation sets
datagen = ImageDataGenerator(rescale=1./255, validation_split=0.2)

# Generate batches of training data
train_generator = datagen.flow_from_directory(
    directory=dataset_path,
    target_size=(img_size, img_size),  # Resize images to 224x224
    batch_size=batch_size,  # Number of images to return in each batch
    class_mode='categorical',  # For classification with multiple classes
    subset='training'  # Use this subset of data for training
)

# Generate batches of validation data
val_generator = datagen.flow_from_directory(
    directory=dataset_path,
    target_size=(img_size, img_size),  # Resize images to 224x224
    batch_size=batch_size,  # Number of images to return in each batch
    class_mode='categorical',  # For classification with multiple classes
    subset='validation'  # Use this subset of data for validation
)

# Load the VGG16 model pre-trained on ImageNet (with the top classification layer removed)
base_model = VGG16(weights='imagenet', include_top=False, input_shape=(img_size, img_size, 3))

# Freeze the pre-trained layers so they dont change during training
for layer in base_model.layers:
    layer.trainable = False

# Add new layers on top of the VGG16 model
x = base_model.output  # Get the output from the base model
x = Flatten()(x)  # Flatten the output to a 1D array
x = Dense(256, activation='relu')(x)  # Add a fully connected layer with 256 units
x = Dense(train_generator.num_classes, activation='softmax')(x)  # Add an output layer with softmax activation

# Create a new model with the modified layers
model = Model(inputs=base_model.input, outputs=x)

# Compile the model with the Adam optimizer and categorical crossentropy loss function
model.compile(optimizer=Adam(), loss='categorical_crossentropy', metrics=['accuracy'])

# Set up callbacks to stop training early if the model stops improving and to save the best model
early_stopping = EarlyStopping(monitor='val_loss', patience=3, restore_best_weights=True)
checkpoint = ModelCheckpoint('final_trained_vgg16_modelfor_rice_breed_classification.h5', monitor='val_loss', save_best_only=True)

# Train the model with the training and validation data
history = model.fit(
    train_generator,  # Data for training
    validation_data=val_generator,  # Data for validation
    epochs=10,  # Number of times to go through the entire dataset
    callbacks=[early_stopping, checkpoint]  # Callbacks to monitor training
)

# Plot the training and validation accuracy and loss
plt.figure(figsize=(12, 5))  # Set the size of the figure

# Plot accuracy
plt.subplot(1, 2, 1)  # Create a subplot for accuracy
plt.plot(history.history['accuracy'])  # Plot training accuracy
plt.plot(history.history['val_accuracy'])  # Plot validation accuracy
plt.title('Model accuracy')  # Title of the accuracy plot
plt.xlabel('Epoch')  # Label for the x-axis
plt.ylabel('Accuracy')  # Label for the y-axis
plt.legend(['Train', 'Validation'], loc='upper left')  # Legend to distinguish lines

# Plot loss
plt.subplot(1, 2, 2)  # Create a subplot for loss
plt.plot(history.history['loss'])  # Plot training loss
plt.plot(history.history['val_loss'])  # Plot validation loss
plt.title('Model loss')  # Title of the loss plot
plt.xlabel('Epoch')  # Label for the x-axis
plt.ylabel('Loss')  # Label for the y-axis
plt.legend(['Train', 'Validation'], loc='upper left')  # Legend to distinguish lines

plt.show()  # Show the plots
